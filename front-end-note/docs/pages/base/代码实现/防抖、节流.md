# 防抖、节流

```javascript
function debounce(fn, delay) {
  let timer = null; //借助闭包
  return function () {
    if (timer) {
      clearTimeout(timer);
      timer = setTimeOut(fn, delay);
    } else {
      timer = setTimeOut(fn, delay);
    }
  };
}

function throttle(fn, delay = 300) {
  let timer = null;
  return function () {
    if (timer) {
      return;
    }
    timer = setTimeout(() => {
      fn.apply(this, arguments);
      timer = null;
    }, delay);
  };
}
```
